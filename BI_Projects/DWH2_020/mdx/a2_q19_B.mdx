-- Business Question Q19 - MDX for Student B (NEW)
-- For 2023 and 2024, compare Missing Days by Region (Year-over-Year).
-- Return Regions on rows and two columns for 2023 and 2024.

SELECT
  NON EMPTY
    CROSSJOIN(
      { [Measures].[Missing Days] },
      {
        [dim_timemonth].[Time].[year_num].&[2023],
        [dim_timemonth].[Time].[year_num].&[2024]
      }
    ) ON COLUMNS,
  NON EMPTY [dim_city].[Geo].[region_name].Members ON ROWS
FROM [AirQ Cube]
